Design an algorithm to print all permutations of a string. For simplicity, assume all characters are unique.
Test String: abcdefg
Case “a” --> {a}
Case “ab” --> {ab, ba}
Case “abc” --> ?

 

Two gains:

1. Array to List, use  Arrays.asList(array) , then it is able to be added to another list, e.g., list.addAll(Arrays.asList(array))

2. List to Array, (for example, cast to String array) use list.toArray(new String[list.size()])

 

[code lang="Java"]

import java.util.Arrays;
import java.util.LinkedList;
public class Main {

    public static void main(String[] args) {
        String input =&quot;abcd&quot;;
        String[] output=recpermutate(input);
        for (String element :output){
            System.out.print(element+&quot;\t&quot;);
        }
    }
    
    static String[] recpermutate(String s){
        if (s.length()==1){
            return new String[]{s};
        }
        else{
            char last=s.charAt(s.length()-1);
            String[] intermOutput=recpermutate(s.substring(0,s.length()-1));
            LinkedList&lt;String&gt; output = new LinkedList&lt;String&gt;();
            for (String element : intermOutput) {
                String[] newBatch= insertAllPositions(last, element);
                output.addAll(Arrays.asList(newBatch));
            }
            return output.toArray(new String[output.size()]);
        }
    }

    static String[] insertAllPositions(char a, String s){
        LinkedList&lt;String&gt; batch = new LinkedList&lt;String&gt;();
        for(int i=s.length();i&gt;=0;--i){
            String newString=s.substring(0, i).concat(String.valueOf(a)).concat(s.substring(i, s.length()));
            batch.add(newString);
        }
        return  batch.toArray(new String[batch.size()]);
    }
}
